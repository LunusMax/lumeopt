\begin{algorithm}[H]
\caption{Algoritmo Guloso com estratégia Best Improvement}
\begin{algorithmic}

\State Obter lista de jogadores candidatos
\State Inicializar conjunto de contratações como vazio
\State Calcular pontuação atual com conjunto vazio
\State Inicializar custo atual igual a zero

\While{verdadeiro}
    \State Inicializar melhor jogador como vazio
    \State Inicializar melhor pontuação como vazia

    \For{cada jogador candidato $p$}
        \If{$p$ ainda não foi selecionado e cabe no orçamento}
            \State Calcular a nova pontuação prevista ao adicionar $p$

            \If{não existir melhor jogador ou a nova pontuação for maior que a melhor pontuação}
                \State Armazenar $p$ como melhor jogador da iteração
                \State Armazenar a nova pontuação como melhor pontuação da iteração
            \EndIf
        \EndIf
    \EndFor

    \If{existe melhor jogador e a melhor pontuação for maior que a pontuação atual}
        \State Adicionar o melhor jogador ao conjunto de contratações
        \State Atualizar a pontuação atual
        \State Atualizar o custo atual
    \Else
        \State \textbf{break}
    \EndIf
\EndWhile

\State \Return conjunto de contratações, custo atual e pontuação prevista

\end{algorithmic}
\end{algorithm}
