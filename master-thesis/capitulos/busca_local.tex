\begin{algorithm}[htbp]
\caption{Busca local com restrição de orçamento (substituição unitária e melhoria estrita)}
\label{alg:busca_local_budget}
\DontPrintSemicolon
\SetKwInOut{Input}{Entrada}
\SetKwInOut{Output}{Saída}

\Input{$team$, $season$, $player\_attributes$, $player\_attributes\_PCA$, $points$, $size$, $budget$, $df\_Players$}
\Output{$X^\star$ (jogadores contratados) e $f(X^\star)$ (pontuação prevista)}

\Fn{preve}{$X$}{
  $team\_df \leftarrow \texttt{create\_team\_pca\_dataframe}(\ldots)$\;
  \If{$team\_df.\text{Over\_Budget} = \text{true}$}{
    \Return{$-9999$}\;
  }
  \Return{$\texttt{preve\_pontos}(\ldots)$}\;
}

$N \leftarrow$ lista de jogadores únicos em $player\_attributes[\text{Player}]$\;
$X^\star \leftarrow \varnothing$\;
$best \leftarrow -\infty$\;

\tcp{Geração de solução inicial viável}
\For{$t \leftarrow 1$ \KwTo $500$}{
  $X \leftarrow$ amostra aleatória sem reposição de $N$ com tamanho $size$\;
  $team\_df \leftarrow \texttt{create\_team\_pca\_dataframe}(\ldots)$\;
  \If{$team\_df.\text{Over\_Budget} = \text{false}$}{
    $X^\star \leftarrow X$\;
    $best \leftarrow preve(X^\star)$\;
    \KwBreak\;
  }
}
\If{$X^\star = \varnothing$}{
  \KwRet{erro: nenhuma combinação inicial viável encontrada}\;
}

\tcp{Busca local: vizinhança por substituição unitária; critério de aceitação por melhoria estrita}
$improved \leftarrow \text{true}$\;
\While{$improved$}{
  $improved \leftarrow \text{false}$\;
  \For{$i \leftarrow 1$ \KwTo $size$}{
    \ForEach{$p \in N$}{
      \If{$p \notin X^\star$}{
        $X' \leftarrow X^\star$\;
        substituir o $i$-ésimo elemento de $X'$ por $p$\;
        $val \leftarrow preve(X')$\;
        \If{$val > best$}{
          $X^\star \leftarrow X'$\;
          $best \leftarrow val$\;
          $improved \leftarrow \text{true}$\;
          \KwBreak\;
        }
      }
    }
    \If{$improved$}{\KwBreak}
  }
}

\Return{$X^\star, best$}\;
\end{algorithm}
