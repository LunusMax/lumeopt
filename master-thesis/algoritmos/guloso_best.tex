\begin{algorithm}[H]
\caption{Algoritmo Guloso Best-Improvement para Seleção de Jogadores}
\begin{algorithmic}[1]

\Function{GulosoBest}{$team, season, player\_attributes, player\_attributes\_PCA, points, size, budget$}

\State \textbf{Função} $\textsc{PreviousSeason}(s)$
\State \hspace{0.5cm} separar $s$ em $y_1$ e $y_2$
\State \hspace{0.5cm} \Return $(y_1 - 1)-(y_2 - 1)$

\State $prev\_season \gets \textsc{PreviousSeason}(season)$

\State $player\_names \gets$ jogadores únicos em $player\_attributes$ com $Season = prev\_season$

\State $current\_team\_df \gets$ registros de $player\_attributes$ tais que
\State \hspace{0.5cm} $Team = team$ e $Season = season$

\State $current\_team\_players \gets$ conjunto de jogadores únicos em $current\_team\_df$

\State definir coluna de minutagem como $Min.$ ou $Min$

\State $current\_team\_roster \gets$ jogadores únicos em $current\_team\_df$

\State $lowest\_minutes\_players \gets$ jogadores de $current\_team\_df$ ordenados por minutagem crescente

\State \textbf{Função} $\textsc{Preve}(new\_players)$
\State \hspace{0.5cm} $n\_remove \gets |new\_players|$
\State \hspace{0.5cm} $removed\_players \gets$ primeiros $n\_remove$ jogadores de $lowest\_minutes\_players$
\State \hspace{0.5cm} $simulated\_roster \gets$ jogadores de $current\_team\_roster$ que não pertencem a $removed\_players$, unidos a $new\_players$
\State \hspace{0.5cm} \Return $\textsc{preve\_pontos}(team, season, simulated\_roster,$
\State \hspace{1.7cm} $player\_attributes, player\_attributes\_PCA, points)$

\State $current\_new\_players \gets \emptyset$
\State $current\_score \gets \textsc{Preve}(current\_new\_players)$

\State $improved \gets$ verdadeiro

\For{$i = 1$ até $size$}
    \If{$improved =$ falso}
        \State \textbf{break}
    \EndIf

    \State $improved \gets$ falso
    \State $best\_p \gets$ nulo
    \State $best\_price \gets$ nulo
    \State $best\_score \gets current\_score$

    \For{cada jogador $p$ em $player\_names$}

        \If{$p \in current\_new\_players$}
            \State \textbf{continue}
        \EndIf

        \If{$p \in current\_team\_players$}
            \State \textbf{continue}
        \EndIf

        \State $row \gets$ registro de $player\_attributes$ tal que
        \State \hspace{0.5cm} $Player = p$ e $Season = prev\_season$

        \If{$row$ vazio}
            \State \textbf{continue}
        \EndIf

        \State $p\_price \gets row.Valor\_Mercado$

        \If{$p\_price > budget$}
            \State \textbf{continue}
        \EndIf

        \State $prev \gets \textsc{Preve}(current\_new\_players \cup \{p\})$

        \If{$prev > best\_score$}
            \State $best\_score \gets prev$
            \State $best\_p \gets p$
            \State $best\_price \gets p\_price$
        \EndIf

    \EndFor

    \If{$best\_p \neq$ nulo}
        \State adicionar $best\_p$ em $current\_new\_players$
        \State $current\_score \gets best\_score$
        \State $budget \gets budget - best\_price$
        \State $improved \gets$ verdadeiro
    \EndIf

\EndFor

\State \Return $current\_new\_players, current\_score, budget$

\EndFunction

\end{algorithmic}
\end{algorithm}
