\begin{algorithm}
\caption{Algoritmo Guloso com estratégia Best Improvement}
\begin{algorithmic}

\State Obter jogadores candidatos da temporada anterior
\State Obter elenco atual do time alvo
\State Ordenar jogadores do elenco atual por menor minutagem

\State $S \gets \emptyset$ \Comment{conjunto de contratações}
\State calcular pontuação inicial

\For{$i = 1$ até $size$}

    \State $best\_player \gets \emptyset$
    \State $best\_score \gets$ pontuação atual

    \For{cada jogador candidato $p$}
    
        \If{$p$ não pertence ao elenco atual e cabe no orçamento}
        
            \State Simular substituição do jogador de menor minutagem por $p$
            \State Calcular pontuação prevista
            
            \If{pontuação $>$ $best\_score$}
                \State atualizar $best\_player$ e $best\_score$
            \EndIf
            
        \EndIf
        
    \EndFor

    \If{$best\_player \neq \emptyset$}
        \State adicionar $best\_player$ ao conjunto de contratações
        \State atualizar orçamento
        \State atualizar pontuação
    \Else
        \State \textbf{break}
    \EndIf

\EndFor

\State \Return contratações e pontuação prevista

\end{algorithmic}
\end{algorithm}
