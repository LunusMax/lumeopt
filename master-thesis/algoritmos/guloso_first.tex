\begin{algorithm}
\caption{Algoritmo Guloso com estratégia First Improvement}
\begin{algorithmic}

\State Obter lista de jogadores candidatos da temporada anterior
\State Obter elenco atual do time alvo
\State Ordenar jogadores do elenco atual por menor minutagem

\State $S \gets \emptyset$ \Comment{novas contratações}
\State calcular pontuação atual

\For{$i = 1$ até $size$}
    \For{cada jogador candidato $p$}
    
        \If{$p$ não pertence ao elenco atual e cabe no orçamento}
        
            \State Simular substituição do jogador de menor minutagem por $p$
            \State Calcular nova pontuação prevista
            
            \If{pontuação melhora}
                \State adicionar $p$ em $S$
                \State atualizar orçamento
                \State atualizar pontuação
                \State \textbf{break} \Comment{first improvement}
            \EndIf
            
        \EndIf
        
    \EndFor
    
    \If{nenhuma melhora encontrada}
        \State \textbf{break}
    \EndIf
    
\EndFor

\State \Return conjunto de contratações e pontuação prevista

\end{algorithmic}
\end{algorithm}
