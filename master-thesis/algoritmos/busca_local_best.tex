\begin{algorithm}
\caption{Busca local com estratégia Best Improvement}
\begin{algorithmic}

\State Obter elenco atual do time alvo
\State Ordenar elenco atual por menor minutagem
\State Selecionar candidatos da temporada anterior
\State Gerar uma solução inicial viável respeitando o orçamento
\State Calcular a pontuação inicial da solução

\While{houver melhoria}
    \State Inicializar a melhor solução vizinha com a solução atual
    \For{cada posição de contratação $i$}
        \For{cada jogador candidato $p$}
            \If{$p$ não pertence ao elenco atual, não está na solução e cabe no orçamento}
                \State Substituir temporariamente o jogador da posição $i$ por $p$
                \State Calcular a nova pontuação prevista
                \If{a pontuação melhorar a melhor vizinha encontrada}
                    \State Atualizar a melhor solução vizinha
                \EndIf
            \EndIf
        \EndFor
    \EndFor
    \If{foi encontrada uma vizinhança melhor}
        \State Aceitar a melhor troca encontrada
        \State Atualizar orçamento e solução atual
    \Else
        \State \textbf{break}
    \EndIf
\EndWhile

\State \Return solução final, pontuação prevista e orçamento restante

\end{algorithmic}
\end{algorithm}
