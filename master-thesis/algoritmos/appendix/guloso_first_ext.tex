\begin{breakablealgorithm}[Algoritmo Guloso First-Improvement para Seleção de Jogadores]
\label{alg:guloso_first}
\begin{algorithmic}[1]

\Function{GulosoFirst}{$team, season, player\_attributes, player\_attributes\_PCA, points, size, budget$}

\Statex \textbf{Função auxiliar} $\textsc{PreviousSeason}(s)$:
\State separar $s$ em $y_1$ e $y_2$
\State \Return $(y_1 - 1)-(y_2 - 1)$

\State $prev\_season \gets \textsc{PreviousSeason}(season)$

\State $player\_names \gets$ jogadores únicos em $player\_attributes$ com $Season = prev\_season$

\State $current\_team\_df \gets$ registros de $player\_attributes$ tais que $Team = team$ e $Season = season$

\State $current\_team\_players \gets$ conjunto de jogadores únicos em $current\_team\_df$

\State definir coluna de minutagem como $Min.$ ou $Min$

\State $current\_team\_roster \gets$ jogadores únicos em $current\_team\_df$

\State $lowest\_minutes\_players \gets$ jogadores de $current\_team\_df$ ordenados por minutagem crescente

\Statex \textbf{Função auxiliar} $\textsc{Preve}(new\_players)$:
\State $n\_remove \gets |new\_players|$
\State $removed\_players \gets$ primeiros $n\_remove$ jogadores de $lowest\_minutes\_players$
\State $simulated\_roster \gets$ jogadores de $current\_team\_roster$ que não pertencem a $removed\_players$, unidos a $new\_players$
\State \Return $\textsc{preve\_pontos}(team, season, simulated\_roster, player\_attributes,$
\Statex \hspace{\algorithmicindent} $player\_attributes\_PCA, points)$

\State $current\_new\_players \gets \emptyset$
\State $current\_score \gets \textsc{Preve}(current\_new\_players)$

\State $improved \gets$ verdadeiro

\For{$i = 1$ até $size$}
    \If{$improved =$ falso}
        \State \textbf{interromper o laço}
    \EndIf

    \State $improved \gets$ falso

    \For{cada jogador $p$ em $player\_names$}

        \If{$p \in current\_new\_players$}
            \State passar ao próximo jogador
        \EndIf

        \If{$p \in current\_team\_players$}
            \State passar ao próximo jogador
        \EndIf

        \State $row \gets$ registro de $player\_attributes$ tal que $Player = p$ e $Season = prev\_season$

        \If{$row$ vazio}
            \State passar ao próximo jogador
        \EndIf

        \State $p\_price \gets row.Valor\_Mercado$

        \If{$p\_price > budget$}
            \State passar ao próximo jogador
        \EndIf

        \State $prev \gets \textsc{Preve}(current\_new\_players \cup \{p\})$

        \If{$prev > current\_score$}
            \State adicionar $p$ em $current\_new\_players$
            \State $current\_score \gets prev$
            \State $budget \gets budget - p\_price$
            \State $improved \gets$ verdadeiro
            \State \textbf{interromper o laço}
        \EndIf

    \EndFor
\EndFor

\State \Return $current\_new\_players, current\_score, budget$

\EndFunction

\end{algorithmic}
\end{breakablealgorithm}
