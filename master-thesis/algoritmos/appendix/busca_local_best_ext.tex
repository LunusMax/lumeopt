\begin{breakablealgorithm}[Busca Local Best-Improvement para Seleção de Jogadores]
\label{alg:busca_local_best}
\begin{algorithmic}[1]

\Function{BuscaLocalBest}{$team, season, player\_attributes, player\_attributes\_PCA, points, size, budget$}

\Statex \textbf{Função auxiliar} $\textsc{PreviousSeason}(s)$:
\State separar $s$ em $y_1$ e $y_2$
\State \Return $(y_1 - 1)-(y_2 - 1)$

\State $prev\_season \gets \textsc{PreviousSeason}(season)$

\State $current\_team\_df \gets$ registros de $player\_attributes$ tais que $Team = team$ e $Season = season$

\State $current\_team\_players \gets$ conjunto de jogadores únicos em $current\_team\_df$

\State definir coluna de minutagem como $Min.$ ou $Min$

\State $current\_team\_roster \gets$ jogadores únicos em $current\_team\_df$

\State $lowest\_minutes\_players \gets$ jogadores de $current\_team\_df$ ordenados por minutagem crescente

\Statex \textbf{Função auxiliar} $\textsc{Preve}(new\_players)$:
\State $n\_remove \gets |new\_players|$
\State $removed\_players \gets$ primeiros $n\_remove$ jogadores de $lowest\_minutes\_players$
\State $simulated\_roster \gets$ jogadores de $current\_team\_roster$ que não pertencem a $removed\_players$, unidos a $new\_players$
\State \Return $\textsc{preve\_pontos}(team, season, simulated\_roster, player\_attributes,$
\Statex \hspace{\algorithmicindent} $player\_attributes\_PCA, points)$

\State $base\_players \gets$ jogadores únicos em $player\_attributes$ com $Season = prev\_season$

\State $player\_names \gets$ amostra aleatória de até $100$ jogadores de $base\_players$

\State $price\_map \gets$ dicionário que associa cada jogador de $prev\_season$ ao seu $Valor\_Mercado$

\State $feasible\_pool \gets$ jogadores em $player\_names$ tais que não pertencem a $current\_team\_players$ e existem em $price\_map$

\State embaralhar $feasible\_pool$

\State $current\_new\_players \gets \emptyset$
\State $remaining\_budget \gets budget$

\For{cada jogador $p$ em $feasible\_pool$}
    \If{$|current\_new\_players| \geq size$}
        \State \textbf{break}
    \EndIf

    \State $p\_price \gets price\_map[p]$

    \If{$p\_price \leq remaining\_budget$}
        \State adicionar $p$ em $current\_new\_players$
        \State $remaining\_budget \gets remaining\_budget - p\_price$
    \EndIf
\EndFor

\State $current\_score \gets \textsc{Preve}(current\_new\_players)$

\State $improvement \gets$ verdadeiro

\While{$improvement =$ verdadeiro}
    \State $improvement \gets$ falso
    \State $best\_score \gets current\_score$
    \State $best\_players \gets$ nulo
    \State $best\_budget \gets$ nulo

    \For{$i = 1$ até $|current\_new\_players|$}
        \For{cada jogador $p$ em $player\_names$}

            \If{$p \in current\_new\_players$}
                \State \textbf{continue}
            \EndIf

            \If{$p \in current\_team\_players$}
                \State \textbf{continue}
            \EndIf

            \If{$p \notin price\_map$}
                \State \textbf{continue}
            \EndIf

            \State $p\_price \gets price\_map[p]$
            \State $old\_player \gets current\_new\_players[i]$
            \State $old\_price \gets price\_map[old\_player]$ ou $0$ se ausente

            \If{$p\_price > remaining\_budget + old\_price$}
                \State \textbf{continue}
            \EndIf

            \State $new\_players \gets$ cópia de $current\_new\_players$
            \State substituir o jogador da posição $i$ por $p$

            \State $new\_score \gets \textsc{Preve}(new\_players)$

            \If{$new\_score > best\_score$}
                \State $best\_score \gets new\_score$
                \State $best\_players \gets new\_players$
                \State $best\_budget \gets remaining\_budget + old\_price - p\_price$
            \EndIf

        \EndFor
    \EndFor

    \If{$best\_players \neq$ nulo}
        \State $current\_new\_players \gets best\_players$
        \State $current\_score \gets best\_score$
        \State $remaining\_budget \gets best\_budget$
        \State $improvement \gets$ verdadeiro
    \EndIf

\EndWhile

\State \Return $current\_new\_players, current\_score, remaining\_budget$

\EndFunction

\end{algorithmic}
\end{breakablealgorithm}
