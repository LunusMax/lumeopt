\begin{algorithm}[H]
\caption{Heurística Gulosa para o TSP (Vizinho Mais Próximo)}
\label{alg:tsp_greedy}

\KwIn{Conjunto de cidades $V$, matriz de distâncias $d(i,j)$}
\KwOut{Ciclo Hamiltoniano $T$}

Escolher cidade inicial $v_0$\;

$T \leftarrow [v_0]$\;

$U \leftarrow V \setminus \{v_0\}$ \tcp*{cidades ainda não visitadas}

$v \leftarrow v_0$\;

\While{$U \neq \emptyset$}{
    escolher $u \in U$ que minimize $d(v,u)$\;
    adicionar $u$ ao final de $T$\;
    $U \leftarrow U \setminus \{u\}$\;
    $v \leftarrow u$\;
}

adicionar $v_0$ ao final de $T$ \tcp*{retorna à cidade inicial}

\Return{$T$}

\end{algorithm}
